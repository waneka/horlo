    'use client'
    import { useMemo } from 'react'
    import { useWatchStore } from '@/store/watchStore'
    import { filterWatches } from '@/lib/filtering'
    import { FilterBar } from '@/components/filters/FilterBar'

      watches: Watch[]
      preferences: UserPreferences

    export function CollectionView({ watches, preferences }: CollectionViewProps) {
      const filters = useWatchStore((s) => s.filters)
      const filtered = useMemo(() => filterWatches(watches, filters), [watches, filters])
      const maxPrice = useMemo(
        () => Math.max(0, ...watches.map((w) => w.marketPrice ?? 0)),
        [watches]
      )
      return (
        <>
          <FilterBar maxPrice={maxPrice} />
          <WatchGrid watches={filtered} collection={watches} preferences={preferences} />
        </>
      )
    }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → Next.js server | Page requests now run through Server Component data fetching; auth enforced at `getCurrentUser()` call |
| Server Component → DAL | Every DAL call scoped by `userId` from session (Phase 4 D-02) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-03-01 | E (Elevation of Privilege) / I (Information Disclosure) | Server Component data fetch | mitigate | Every converted page calls `getCurrentUser()` FIRST, then passes `user.id` to `getWatchesByUser` / `getPreferencesByUser`. `getCurrentUser()` throws UnauthorizedError if no session (Phase 4 D-01). The DAL re-verifies userId scoping (Phase 4 D-04). No client-supplied ID is ever trusted. |
| T-05-03-02 | I | props passed from Server to Client | accept | Watch[] and UserPreferences are plain serializable objects; no secrets or tokens embedded. Passing them as props crosses the Server→Client boundary but contains only data the authenticated user already owns. |
| T-05-03-03 | T (Tampering) | client-side filterWatches | accept | Filtering happens client-side in CollectionView. A malicious client could skip the filter, but the data available to filter is already scoped to the user's own collection by the DAL — no data exposure beyond what they own. Filter bypass just shows more of their own watches. |
| T-05-03-04 | S (Spoofing) | Pitfall 1 (re-adding 'use client') | mitigate | Acceptance criteria enforce `grep "'use client'" src/app/page.tsx src/app/insights/page.tsx src/app/preferences/page.tsx` returns no output. If an executor bypasses by re-adding `'use client'` to fix a hook error, the grep gate fails and the plan does not complete. |
</threat_model>
