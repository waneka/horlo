const UUID_RE = /^[0-9a-f-]{36}$/i

export function buildWearPhotoPath(
  userId: string,
  wearEventId: string,
): string {
  if (!userId) {
    throw new TypeError('userId required')
  }
  if (!UUID_RE.test(wearEventId)) {
    throw new TypeError('wearEventId must be a UUID')
  }
  return `${userId}/${wearEventId}.jpg`
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Phase 77 client code ‚Üí `buildWearVideoPath` / `buildWearPosterPath` | Client invokes these to compute the upload path BEFORE the client-direct `supabase.storage.upload(...)` call. The values are NOT trusted by the server ‚Äî Plan 03's Server Action recomputes the same strings from `getCurrentUser().id` independently. |
| `buildWearVideoPath` return value ‚Üí Storage upload | Client uploads the blob to Storage at the returned path. Storage RLS enforces folder-prefix ownership (`(storage.foldername(name))[1] = auth.uid()::text`) ‚Äî a tampered client value would either match the user's own folder (no IDOR) or be rejected by RLS at upload time. |
