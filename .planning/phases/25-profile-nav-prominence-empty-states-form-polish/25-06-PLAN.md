```typescript
// src/lib/hooks/useFormFeedback.ts
interface UseFormFeedbackOptions {
  dialogMode?: boolean
}
interface UseFormFeedbackReturn<T> {
  pending: boolean
  state: 'idle' | 'pending' | 'success' | 'error'
  message: string | null
  dialogMode: boolean
  run: (
    action: () => Promise<ActionResult<T>>,
    opts?: { successMessage?: string; errorMessage?: string },
  ) => Promise<void>
  reset: () => void
}
export function useFormFeedback<T = unknown>(options?: UseFormFeedbackOptions): UseFormFeedbackReturn<T>
```

## Trust Boundaries

| Boundary | Description |

| Client form → Server Action | Each form invokes its respective Server Action via the `useFormFeedback` hook's `run()`. Server Actions enforce their own auth and validation; this plan does not introduce new trust gaps. |
| Client → Sonner toast surface | `toast.success(message)` and `toast.error(message)` render text content; React + Sonner escape strings; no XSS surface. |
| Client → react-dom useFormStatus | useFormStatus is a stable React 19 / Next 16 primitive; CSRF for Server Action invocations is handled by Next's built-in middleware. |
