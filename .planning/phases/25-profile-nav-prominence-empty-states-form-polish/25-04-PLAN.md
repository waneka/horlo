    type ExtractErrorCategory =
      | 'host-403'
      | 'structured-data-missing'
      | 'LLM-timeout'
      | 'quota-exceeded'
      | 'generic-network'

    const CATEGORY_COPY: Record<ExtractErrorCategory, string> = {

      'quota-exceeded': "Extraction service is busy. Try again in a few minutes.",
      'generic-network': "Couldn't reach that URL. Check the link and try again.",
    }

    function categorizeExtractionError(err: unknown): ExtractErrorCategory {
      if (err instanceof Error) {
        // host-403: thrown by fetchAndExtract as "Failed to fetch URL: 403 ..." or any 4xx forbidden status
        if (/Failed to fetch URL:\s*403\b/.test(err.message)) return 'host-403'
        // LLM-timeout: AbortError name or "timeout" in the message
        if (err.name === 'AbortError' || /timeout/i.test(err.message)) return 'LLM-timeout'
      }
      // quota-exceeded: Anthropic SDK 429 (rate_limit_error OR overloaded_error per D-13)
      // Handle both `instanceof APIError` and the generic shape with .status === 429
      // (SDK may wrap or rethrow; defensive check)
      const errAny = err as { status?: unknown; name?: string }
      if (typeof errAny?.status === 'number' && errAny.status === 429) return 'quota-exceeded'
      if (errAny?.name === 'RateLimitError' || errAny?.name === 'APIError' && (errAny as { status?: number }).status === 429) {
        return 'quota-exceeded'
      }
      return 'generic-network'
    }

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Untrusted external URL → server fetch | `fetchAndExtract` follows the user-supplied URL through `safeFetch` (existing SSRF guard). Errors from the external host (403, network) are caught here. |
| External LLM (Anthropic) → server | LLM responses can throw `Anthropic.APIError`. SDK errors include status codes and provider names; sanitization required before client emit. |
| Server response → client | The route's response body is the only attack surface for info disclosure. Sanitization is enforced by always returning `CATEGORY_COPY[category]` (not raw error.message). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-25-04-01 | I (Information disclosure via error message) | route.ts catch block response body | mitigate | Per orchestrator security note: "URL-extract error taxonomy: information disclosure via error messages — error copy is locked (D-15) and must NOT include raw stack traces, internal URLs, or LLM provider names visible to the client". Implementation: response `error` field is ALWAYS `CATEGORY_COPY[category]` (compile-time constant from D-15 LOCKED list). NEVER `err.message`, `err.stack`, `String(err)`. The raw error is logged server-side via `console.error` only. Grep gate in acceptance criteria: response body contains no "anthropic" / "claude" / "stack". |
| T-25-04-02 | I (Information disclosure via category enum) | category field in response | accept | The category enum values themselves (e.g., `'quota-exceeded'`) are user-facing and don't leak provider details. "quota-exceeded" is a generic concept, not Anthropic-specific. The 5 categories are pre-published in this PLAN; revealing them is no leak. |
| T-25-04-03 | T (Tampering with category in client response handling) | AddWatchFlow consumes `data.category` from response | mitigate | TypeScript narrows `data.category` via the FlowState union type. Defensive fallback (`?? 'generic-network'`) handles any unexpected value gracefully.
