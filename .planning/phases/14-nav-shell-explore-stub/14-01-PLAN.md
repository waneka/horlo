    ```ts
    export const PUBLIC_PATHS = [
      '/login',
      '/signup',
      '/forgot-password',
      '/reset-password',
      '/auth',
    ] as const

    export type PublicPath = typeof PUBLIC_PATHS[number]

    /**
     * True when `pathname` matches any PUBLIC_PATH as an exact match, as a
     * prefix followed by '/', or as a prefix followed by '?' (query string).
     *
     * Mirrors the proxy's `startsWith(p)` semantics while explicitly rejecting
     * collisions like `/loginfoo` or `/authentication`. All five public paths
     * are short and well-known; no real route uses these prefixes for other
     * purposes (verified RESEARCH §Pattern 3).
     */
    export function isPublicPath(pathname: string): boolean {
      return PUBLIC_PATHS.some(
        (p) =>
          pathname === p ||
          pathname.startsWith(`${p}/`) ||
          pathname.startsWith(`${p}?`),
      )
    }
    ```
