KEEP the existing debounce useEffect (lines 59-62) and URL sync useEffect (lines 65-71) BYTE-IDENTICAL. REPLACE the fetch useEffect (lines 73-114) with a new tab-dispatched version. EXPAND `UseSearchState` interface with per-tab slices.

```typescript
'use client'

import { useEffect, useState } from 'react'
import { useRouter, useSearchParams } from 'next/navigation'

import {
  searchPeopleAction,
  searchWatchesAction,
  searchCollectionsAction,
} from '@/app/actions/search'
import type {
  SearchProfileResult,
  SearchCatalogWatchResult,
  SearchCollectionResult,
  SearchTab,
} from '@/lib/searchTypes'

const DEBOUNCE_MS = 250
const CLIENT_MIN_CHARS = 2
const ALL_TAB_SECTION_CAP = 5 // SRCH-13

export interface UseSearchState {
  q: string
  setQ: (next: string) => void
  debouncedQ: string
  tab: SearchTab
  setTab: (next: SearchTab) => void
  peopleResults: SearchProfileResult[]
  watchesResults: SearchCatalogWatchResult[]
  collectionsResults: SearchCollectionResult[]
  peopleIsLoading: boolean
  watchesIsLoading: boolean
  collectionsIsLoading: boolean
  peopleHasError: boolean
  watchesHasError: boolean
  collectionsHasError: boolean
}

/**
 * Phase 19 extended search hook.
 *
 * - Debounce + URL sync + 2-char minimum carry forward unchanged from Phase 16.
 * - Pitfall 9 fix: the old `tab !== 'all' && tab !== 'people'` early-return is
 *   removed. Watches and Collections both fire fetches now.
 * - SRCH-14: dependency array gains `tab` so switching tabs while a fetch is
 *   in flight aborts the prior controller and re-dispatches the right action.
 * - D-15: All tab fires 3 actions in parallel via Promise.all. Each section's
 *   slice is set independently. The single AbortController atomically cancels
 *   all 3 in-flight fetches at the network layer (RESEARCH Pattern 3).
 * - SRCH-13: All tab caps each section at 5 results. Per-tab views show
 *   the full 20-result slice (no cap beyond DAL's limit).
 *
 * Per-tab slices keep error/loading state isolated — one section's error
 * does not zero out the others' results.
 */
export function useSearchState(): UseSearchState {
  const router = useRouter()
  const searchParams = useSearchParams()
  const initialQ = searchParams.get('q') ?? ''
  const initialTab = (searchParams.get('tab') as SearchTab | null) ?? 'all'

  const [q, setQ] = useState(initialQ)
  const [debouncedQ, setDebouncedQ] = useState(initialQ)
  const [tab, setTabState] = useState<SearchTab>(initialTab)

  const [peopleResults, setPeopleResults] = useState<SearchProfileResult[]>([])
  const [watchesResults, setWatchesResults] = useState<SearchCatalogWatchResult[]>([])
  const [collectionsResults, setCollectionsResults] = useState<SearchCollectionResult[]>([])

  const [peopleIsLoading, setPeopleIsLoading] = useState(false)
  const [watchesIsLoading, setWatchesIsLoading] = useState(false)
  const [collectionsIsLoading, setCollectionsIsLoading] = useState(false)

  const [peopleHasError, setPeopleHasError] = useState(false)
  const [watchesHasError, setWatchesHasError] = useState(false)
  const [collectionsHasError, setCollectionsHasError] = useState(false)

  // 1. Debounce q → debouncedQ (Phase 16 carry-forward, byte-identical).
  useEffect(() => {
    const t = setTimeout(() => setDebouncedQ(q), DEBOUNCE_MS)
    return () => clearTimeout(t)
  }, [q])

  // 2. URL sync (Phase 16 carry-forward, byte-identical).
  useEffect(() => {
    const params = new URLSearchParams()
    if (debouncedQ.trim().length >= CLIENT_MIN_CHARS) params.set('q', debouncedQ)
    if (tab !== 'all') params.set('tab', tab)
    const qs = params.toString()
    router.replace(qs ? `/search?${qs}` : '/search', { scroll: false })
  }, [debouncedQ, tab, router])

  // 3. Fetch effect — tab-dispatched, per-tab AbortController (SRCH-14).
  useEffect(() => {
    // 2-char client minimum (server-side authoritative, this is defense-in-depth)
    if (debouncedQ.trim().length < CLIENT_MIN_CHARS) {
      setPeopleResults([]); setWatchesResults([]); setCollectionsResults([])
      setPeopleIsLoading(false); setWatchesIsLoading(false); setCollectionsIsLoading(false)
      setPeopleHasError(false); setWatchesHasError(false); setCollectionsHasError(false)
      return
    }

    const controller = new AbortController()
    const fireAll = tab === 'all'

    if (fireAll || tab === 'people') {
      setPeopleIsLoading(true); setPeopleHasError(false)
    }
    if (fireAll || tab === 'watches') {
      setWatchesIsLoading(true); setWatchesHasError(false)
    }
    if (fireAll || tab === 'collections') {
      setCollectionsIsLoading(true); setCollectionsHasError(false)
    }

    void (async () => {
      try {
        if (fireAll) {
          // D-15 parallel fetch
          const [people, watches, collections] = await Promise.all([
            searchPeopleAction({ q: debouncedQ }),
            searchWatchesAction({ q: debouncedQ }),
            searchCollectionsAction({ q: debouncedQ }),
          ])
          if (controller.signal.aborted) return // Pitfall 3 stale-result guard

          // SRCH-13: cap each section at 5
          if (people.success) setPeopleResults(people.data.slice(0, ALL_TAB_SECTION_CAP))
          else { setPeopleHasError(true); setPeopleResults([]) }

          if (watches.success) setWatchesResults(watches.data.slice(0, ALL_TAB_SECTION_CAP))
          else { setWatchesHasError(true); setWatchesResults([]) }

          if (collections.success) setCollectionsResults(collections.data.slice(0, ALL_TAB_SECTION_CAP))
          else { setCollectionsHasError(true); setCollectionsResults([]) }
        } else if (tab === 'people') {
          const result = await searchPeopleAction({ q: debouncedQ })
          if (controller.signal.aborted) return
          if (result.success) setPeopleResults(result.data)
          else { setPeopleHasError(true); setPeopleResults([]) }
        } else if (tab === 'watches') {
          const result = await searchWatchesAction({ q: debouncedQ })
          if (controller.signal.aborted) return
          if (result.success) setWatchesResults(result.data)
          else { setWatchesHasError(true); setWatchesResults([]) }
        } else {
          // tab === 'collections'
          const result = await searchCollectionsAction({ q: debouncedQ })
          if (controller.signal.aborted) return
          if (result.success) setCollectionsResults(result.data)
          else { setCollectionsHasError(true); setCollectionsResults([]) }
        }
      } catch (err) {
        if ((err as Error)?.name === 'AbortError') return
        if (controller.signal.aborted) return
        if (fireAll || tab === 'people') setPeopleHasError(true)
        if (fireAll || tab === 'watches') setWatchesHasError(true)
        if (fireAll || tab === 'collections') setCollectionsHasError(true)
      } finally {
        if (!controller.signal.aborted) {
          if (fireAll || tab === 'people') setPeopleIsLoading(false)
          if (fireAll || tab === 'watches') setWatchesIsLoading(false)
          if (fireAll || tab === 'collections') setCollectionsIsLoading(false)
        }
      }
    })()

    return () => controller.abort()
  }, [debouncedQ, tab])

  const setTab = (next: SearchTab) => setTabState(next)

  return {
    q, setQ, debouncedQ, tab, setTab,
    peopleResults, watchesResults, collectionsResults,
    peopleIsLoading, watchesIsLoading, collectionsIsLoading,
    peopleHasError, watchesHasError, collectionsHasError,
  }
}
```

```tsx
  viewerId: string

/**

export function AllTabResults({
  q,
  viewerId,
  peopleResults,
  watchesResults,
  collectionsResults,
  peopleIsLoading,
  watchesIsLoading,
  collectionsIsLoading,
  setTab,
}: AllTabResultsProps) {
  return (
    <div className="space-y-6">
      <Section
        label="People"
        showSeeAll={peopleResults.length === ALL_TAB_SECTION_CAP}
        onSeeAll={() => setTab('people')}
      >
        {peopleIsLoading ? (
          <SearchResultsSkeleton />
        ) : peopleResults.length === 0 ? (
          <p className="text-sm text-muted-foreground">No matches</p>
        ) : (
          <div className="space-y-2">
            {peopleResults.map((r) => (
              <PeopleSearchRow key={r.userId} result={r} q={q} viewerId={viewerId} />

          <div className="space-y-2">
            {watchesResults.map((r) => (
              <WatchSearchRow key={r.catalogId} result={r} q={q} />
            ))}
          </div>
        )}
      </Section>

      <Section
        label="Collections"
        showSeeAll={collectionsResults.length === ALL_TAB_SECTION_CAP}
```
