### A. Create `src/lib/images.ts` with:
```ts
export const ALLOWED_HOSTS = [
  'hodinkee.com',
  'chrono24.com',
  'watchuseek.com',
  'rolex.com',
  'omega-watches.com',
  'tudorwatch.com',
  'seikowatches.com',
  'grand-seiko.com',
  'wornandwound.com',
  'teddybaldassarre.com',
  'watchesofmayfair.com',
  'cdn.shopify.com',
  'squarespace-cdn.com',
  'images.squarespace-cdn.com',
] as const

export function getSafeImageUrl(url: string | undefined | null): string | null {
  if (!url) return null
  try {
    const { hostname } = new URL(url)
    const isAllowed = ALLOWED_HOSTS.some(
      (h) => hostname === h || hostname.endsWith('.' + h)
    )
    return isAllowed ? url : null
  } catch {
    return null
  }
}
```

import { getSafeImageUrl, ALLOWED_HOSTS } from '@/lib/images'

describe('getSafeImageUrl', () => {
  it('returns null for undefined/null/empty', () => {
    expect(getSafeImageUrl(undefined)).toBeNull()
    expect(getSafeImageUrl(null)).toBeNull()
    expect(getSafeImageUrl('')).toBeNull()
  })

  it('returns null for malformed URLs', () => {
    expect(getSafeImageUrl('not-a-url')).toBeNull()
  })

  it('returns null for disallowed hosts', () => {
    expect(getSafeImageUrl('https://evil.example.com/foo.jpg')).toBeNull()
    expect(getSafeImageUrl('http://localhost/foo')).toBeNull()
    expect(getSafeImageUrl('https://pinterest.com/pin/123')).toBeNull()
  })

  it('returns the URL for exact-host matches', () => {
    expect(getSafeImageUrl('https://hodinkee.com/foo.jpg')).toBe('https://hodinkee.com/foo.jpg')
    expect(getSafeImageUrl('https://cdn.shopify.com/s/foo.jpg')).toBe('https://cdn.shopify.com/s/foo.jpg')
  })

  it('returns the URL for subdomain matches', () => {
    expect(getSafeImageUrl('https://cdn.hodinkee.imgix.net.hodinkee.com/foo.jpg')).toMatch(/hodinkee/)

import { describe, it, expect } from 'vitest'
import { readFileSync, readdirSync, statSync } from 'node:fs'
import { join } from 'node:path'

function walk(dir: string): string[] {
  const out: string[] = []
  for (const name of readdirSync(dir)) {
    const full = join(dir, name)
    const s = statSync(full)
    if (s.isDirectory()) out.push(...walk(full))
    else if (/\.(tsx|jsx)$/.test(name)) out.push(full)
  }
  return out
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client-supplied `watch.imageUrl` -> next/image optimizer | Attacker could inject an arbitrary hostname via the watch form or URL import |
| next/image optimizer -> server fetch | Unlisted host would otherwise be proxied by Next.js optimization endpoint |
| Rendered `<img>` tag -> browser loader | Raw `<img>` tags bypass Next.js optimizer entirely |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-20 | Spoofing | Attacker-supplied image host (`watch.imageUrl = 'https://internal.corp/…'`) | mitigate | Task 1 `remotePatterns` in `next.config.ts` refuses any optimization request whose host is not in the allow-list; enforced server-side by Next.js |
| T-01-21 | Information Disclosure | next/image optimizer acting as an unintended image proxy | mitigate | Same — `remotePatterns` is the documented defense; covered by `npm run build` validation |
| T-01-22 | Tampering | Future PR adds a raw `<img>` tag to watch components, bypassing the helper | mitigate | Task 2 creates `tests/no-raw-img.test.ts` that fails CI if any `<img ` token appears in `WatchCard.tsx` or `WatchDetail.tsx` |
| T-01-23 | Elevation of Privilege | XSS via `<img onerror=...>` | mitigate | React auto-escapes attributes + `next/image` serializes `src` through its optimizer; no user-controlled HTML is rendered |
| T-01-24 | Information Disclosure | Referrer leakage to external hosts | accept | Out of scope for Phase 1; revisit with `referrerpolicy` in a later polish pass |
</threat_model>
