    ```tsx
    import { getWatchesByUser } from '@/data/watches'
    import { getMostRecentWearDates, getAllWearEventsByUser } from '@/data/wearEvents'
    import { getFollowingForProfile } from '@/data/follows'
    import { getTasteOverlapData } from '@/data/follows'
    import { computeTasteOverlap } from '@/lib/tasteOverlap'
    import { wishlistGap } from '@/lib/wishlistGap'
    import { SleepingBeautyCard } from '@/components/home/SleepingBeautyCard'
    import { MostWornThisMonthCard } from '@/components/home/MostWornThisMonthCard'
    import { WishlistGapCard } from '@/components/home/WishlistGapCard'
    import { CommonGroundFollowerCard } from '@/components/home/CommonGroundFollowerCard'
    import { daysSince } from '@/lib/wear'

    const SLEEPING_BEAUTY_THRESHOLD_DAYS = 14

    export async function PersonalInsightsGrid({ viewerId }: { viewerId: string }) {
      const [watches, wearEvents, following] = await Promise.all([
        getWatchesByUser(viewerId),
        getAllWearEventsByUser(viewerId),
        getFollowingForProfile(viewerId),
      ])

      const owned = watches.filter((w) => w.status === 'owned')
      const wishlist = watches.filter((w) => w.status === 'wishlist' || w.status === 'grail')
      if (owned.length === 0) return null   // I-04

      // Sleeping Beauty
      const wearDateMap = await getMostRecentWearDates(viewerId, owned.map((w) => w.id))
      let sleepingBeauty: { watch: typeof owned[number]; days: number } | null = null
      for (const w of owned) {
        const last = wearDateMap.get(w.id)
        const days = last ? daysSince(last) : null
        const effectiveDays = days ?? Number.POSITIVE_INFINITY   // never worn → infinity
        if (effectiveDays > SLEEPING_BEAUTY_THRESHOLD_DAYS) {
          if (!sleepingBeauty || effectiveDays > sleepingBeauty.days) {
            sleepingBeauty = { watch: w, days: Number.isFinite(effectiveDays) ? effectiveDays : 999 }
          }
        }
      }

      // Most worn this month
      const monthStart = new Date()
      monthStart.setDate(1); monthStart.setHours(0, 0, 0, 0)
      const monthStartStr = monthStart.toISOString().slice(0, 10)
      const thisMonthWears = wearEvents.filter((w) => w.wornDate >= monthStartStr)
      const countByWatch = new Map<string, number>()
      for (const w of thisMonthWears) countByWatch.set(w.watchId, (countByWatch.get(w.watchId) ?? 0) + 1)
      let mostWorn: { watch: typeof owned[number]; count: number } | null = null
      for (const [watchId, count] of countByWatch) {
        if (!mostWorn || count > mostWorn.count) {
          const watch = owned.find((w) => w.id === watchId)
          if (watch) mostWorn = { watch, count }
        }
      }

      // Wishlist gap
      const gap = wishlistGap(owned, wishlist)

      // Common Ground follower — pick the following with highest overlap
      let commonGround: { username: string; displayName: string | null; avatarUrl: string | null; sharedCount: number } | null = null
      if (following.length > 0) {
        const scored = await Promise.all(
          following.slice(0, 10).map(async (f) => {
            try {
              const data = await getTasteOverlapData(viewerId, f.userId)
    ```
