    import { describe, it, expect, beforeAll, afterAll } from 'vitest'
    import { db } from '@/db'
    import { users, profiles, profileSettings, follows, watches, activities, wearEvents } from '@/db/schema'
    import { getFeedForUser } from '@/data/activities'
    import { getWearRailForViewer } from '@/data/wearEvents'
    import { getSuggestedCollectors } from '@/data/suggestions'
    import { eq, inArray } from 'drizzle-orm'

    // Gate the entire suite on DB availability.
    describe.skipIf(!process.env.DATABASE_URL)('Phase 10 home privacy end-to-end', () => {
      const ids = {
        V: '00000000-0000-0000-0000-00000000000a',
        A: '00000000-0000-0000-0000-00000000000b',
        B: '00000000-0000-0000-0000-00000000000c',
        C: '00000000-0000-0000-0000-00000000000d',
        D: '00000000-0000-0000-0000-00000000000e',
        E: '00000000-0000-0000-0000-00000000000f',
      }

      beforeAll(async () => {
        // Clean seed
        const allIds = Object.values(ids)
        await db.delete(activities).where(inArray(activities.userId, allIds))
        await db.delete(wearEvents).where(inArray(wearEvents.userId, allIds))
        await db.delete(watches).where(inArray(watches.userId, allIds))
        await db.delete(follows).where(inArray(follows.followerId, allIds))
        await db.delete(profileSettings).where(inArray(profileSettings.userId, allIds))
        await db.delete(profiles).where(inArray(profiles.id, allIds))
        await db.delete(users).where(inArray(users.id, allIds))

        // Seed users + profiles + settings
        await db.insert(users).values(
          Object.entries(ids).map(([k, id]) => ({ id, email: `${k.toLowerCase()}@test.horlo` })),
        )
        await db.insert(profiles).values(
          Object.entries(ids).map(([k, id]) => ({
            id, username: `user_${k.toLowerCase()}`, displayName: k, bio: null, avatarUrl: null,
          })),
        )
        await db.insert(profileSettings).values([
          { userId: ids.V, profilePublic: true, collectionPublic: true, wishlistPublic: true, wornPublic: true },
          { userId: ids.A, profilePublic: true, collectionPublic: true, wishlistPublic: true, wornPublic: true },
          { userId: ids.B, profilePublic: true, collectionPublic: false, wishlistPublic: true, wornPublic: true },
          { userId: ids.C, profilePublic: false, collectionPublic: true, wishlistPublic: true, wornPublic: true },
          { userId: ids.D, profilePublic: true, collectionPublic: true, wishlistPublic: true, wornPublic: true },
          { userId: ids.E, profilePublic: false, collectionPublic: true, wishlistPublic: true, wornPublic: true },
        ])

        // V follows A, B, C (NOT D, E)
        await db.insert(follows).values([
          { followerId: ids.V, followingId: ids.A },
          { followerId: ids.V, followingId: ids.B },
          { followerId: ids.V, followingId: ids.C },
        ])

        // Seed one watch per user for activities to reference
        const watchRows = await db.insert(watches).values(
          Object.entries(ids).map(([k, id]) => ({
            userId: id,
            brand: `Brand-${k}`,
            model: `Model-${k}`,
            status: 'owned' as const,
            movement: 'automatic' as const,
          })),
        ).returning()
        const watchByUser = new Map(watchRows.map((w) => [w.userId, w]))

        const now = new Date()
        const oneHourAgo = new Date(now.getTime() - 60 * 60 * 1000)
        const halfHourAgo = new Date(now.getTime() - 30 * 60 * 1000)
        const today = now.toISOString().slice(0, 10)

        // Activities — V's own + A/B/C/D have 1 watch_added and 1 watch_worn
        const actRows = [ids.V, ids.A, ids.B, ids.C, ids.D].flatMap((uid) => {
          const w = watchByUser.get(uid)!
          return [
            { userId: uid, type: 'watch_added', watchId: w.id, metadata: { brand: w.brand, model: w.model, imageUrl: null }, createdAt: oneHourAgo },
            { userId: uid, type: 'watch_worn', watchId: w.id, metadata: { brand: w.brand, model: w.model, imageUrl: null }, createdAt: halfHourAgo },
          ]
        })
        await db.insert(activities).values(actRows as any)

        // Wear events — all users have today's wear (for WYWT rail)
        await db.insert(wearEvents).values(
          [ids.V, ids.A, ids.B, ids.C, ids.D].map((uid) => {
            const w = watchByUser.get(uid)!
            return { userId: uid, watchId: w.id, wornDate: today }
          }),
        )
      })

      it('feed includes A both events + B watch_worn; excludes B watch_added (collection_public=false) and all C (profile_public=false)', async () => {
        const { rows } = await getFeedForUser(ids.V, null, 50)
        const byUser: Record<string, string[]> = {}
        for (const r of rows) {
          if (r.kind !== 'raw') continue
          ;(byUser[r.userId] ??= []).push(r.type)
        }
        expect(byUser[ids.A]?.sort()).toEqual(['watch_added', 'watch_worn'])
        expect(byUser[ids.B]).toEqual(['watch_worn'])
        expect(byUser[ids.C]).toBeUndefined()
      })

      it('feed excludes non-followed user D', async () => {
        const { rows } = await getFeedForUser(ids.V, null, 50)
        expect(rows.every((r) => r.kind !== 'raw' || r.userId !== ids.D)).toBe(true)
      })

      it("feed excludes viewer's own activities (F-05)", async () => {
        const { rows } = await getFeedForUser(ids.V, null, 50)
        expect(rows.every((r) => r.kind !== 'raw' || r.userId !== ids.V)).toBe(true)
      })

      it('WYWT rail includes V + A + B; excludes C and D', async () => {
        const { tiles } = await getWearRailForViewer(ids.V)
        const userIds = tiles.map((t) => t.userId).sort()
        expect(userIds).toContain(ids.V)
        expect(userIds).toContain(ids.A)
        expect(userIds).toContain(ids.B)
        expect(userIds).not.toContain(ids.C)
        expect(userIds).not.toContain(ids.D)
      })

      it('Suggested Collectors includes D; excludes E (private), A/B/C (already followed), and V (self)', async () => {
        const suggestions = await getSuggestedCollectors(ids.V, 20)
        const userIds = suggestions.map((s) => s.userId)
        expect(userIds).toContain(ids.D)
        expect(userIds).not.toContain(ids.E)
        expect(userIds).not.toContain(ids.A)
        expect(userIds).not.toContain(ids.B)
        expect(userIds).not.toContain(ids.C)
        expect(userIds).not.toContain(ids.V)
      })

      afterAll(async () => {
        const allIds = Object.values(ids)
        await db.delete(activities).where(inArray(activities.userId, allIds))
        await db.delete(wearEvents).where(inArray(wearEvents.userId, allIds))
        await db.delete(watches).where(inArray(watches.userId, allIds))
        await db.delete(follows).where(inArray(follows.followerId, allIds))
        await db.delete(profileSettings).where(inArray(profileSettings.userId, allIds))
        await db.delete(profiles).where(inArray(profiles.id, allIds))
        await db.delete(users).where(inArray(users.id, allIds))
      })
    })
